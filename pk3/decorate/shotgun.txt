Actor CombatShotgun : Weapon Replaces Shotgun
{
	States
	{
		Ready:
			SHTG A 1 A_WeaponReady
			Loop
		Deselect:
			SHTG A 1 A_Lower
			Loop
		Select:
			SHTG A 1 A_Raise
			Loop
		Fire:
			SHTG A 3
			SHTG A 7 A_FireShotgun
			SHTG BC 5
			SHTG D 4
			SHTG CB 5
			SHTG A 3
			SHTG A 7 A_ReFire
			Goto Ready
		AltFire:
			RPUN B 1
			RPUN C 2 A_CustomPunch(25, 1, 4, "MeleePuff")
			RPUN D 8
			RPUN C 7
			Goto Ready
		Flash:
			SHTF A 4 Bright A_Light1
			SHTF B 3 Bright A_Light2
			Goto LightDone
		Spawn:
			SHOT A -1
			Stop
	}
}
